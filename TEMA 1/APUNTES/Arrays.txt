new Array() --> Crea un nuevo array.
    Ejm: 
        frutas = new Array ("fresa", "pera", "manzana")
        frutas = [fresa, pera, manzana]

Array.from() --> Convierte objetos iterables (como strings) en arrays.
    Ejm:
        Array.from("Hola) = ["H", "o", "l", "a"]

Array.of() --> Crea un array a partir de valores pasados como argumentos.
    Ejm:
        Array.of(1, 2, 3) = [1, 2, 3]

push() --> Añade uno o más elementos al final del array.
    Ejm:
        frutas = ["manzana", "pera"];
        frutas.push("uva") = [manzana, pera, uva]

pop() --> Elimina el último elemento y lo devuelve.
    Ejm:
        colores = ["rojo", "verde", "azul"]
        colores.pop() = [rojo, verde] "azul"

unshift() --> Añade uno o más elementos al principio.
    Ejm:
        numeros = [2, 3]
        numeros.unshift(1) = [1, 2, 3]

shift() --> Elimina el primer elemento y lo devuelve.
    Ejm:
        letras = ["a", "b", "c"]
        letras.shift() = [b,c] "a"

splice(x,y) --> Permite añadir o eliminar elementos en una posición concreta (siendo x la posicion) (siendo y=0 para añadir e y=2 para eliminar)
    Ejm:
        meses = ["enero", "marzo", "abril"]
        meses.splice(1,0,"febrero") = ["enero", "febrero", "marzo", "abril"]
        meses.splice(2,1) = ["enero", "febrero", "abril"]

includes() --> Comprueba si un valor existe en el array (devuelve true o false).
    Ejm:
        numeros = [1,2,3]
        numeros.includes(2) = true

indexOf() --> Devuelven la posición de la primera aparición de un elemento.
lastIndexOf() --> Devuelven la posición de la última aparición de un elemento.
    Ejm:
        letras = ["a", "b", "a"]
        letras.indexOf(a) = 0
        letras.lastIndexOf(a) = 2

find() --> Buscan el primer elemento que cumpla una condición.
    Ejm:
        edad = [3,12,21,4,19]
        function mayorEdad(edad) { return edad > 18; }
        edad.find(mayorEdad) = 21

some() --> Comprueba si algún elemento cumple la condición.
    Ejm:
        numero = [1,2,3,-4]
        function negativo(numero) {return numero < 0 }
        numero.some(negativo) = true

every() --> Comprueba si todos los elementos la cumplen.
    Ejm:
        numero = [1,2,3,-4]
        function negativo(numero) {return numero < 0 }
        numero.every(negativo) = false

forEach() --> Ejecuta una función por cada elemento
    Ejm: 
        frutas = ["fresa", "pera", "manzana"]
        frutas.forEach() = fresa // pera // manzana

map() --> Crea un nuevo array aplicando una función a cada elemento.
    Ejm: 
        numeros = [4,9,25]
        numeros.map(Math.sqrt) = 2,3,5

filter() --> Crea un nuevo array con los elementos que cumplen una condición.
    Ejm:
        edad = [3,12,21,4,19]
        function mayorEdad(edad) { return edad > 18; }
        edad.find(mayorEdad) = 21,19

join() -->  Convierte el array en un texto, separando por un delimitador.
    Ejm:
        letras = ["a", "b", "c"]
        letras.join(" - ") = a - b -c

sort() --> Ordena los elementos alfabéticamente o por función personalizada.
    Ejm: 
        num = [12,5,21,2,1]
        num.sort() = 1,12,2,21,5
        num.sort((a,b) => a - b) = 1,2,5,12,21ç

reverse() --> Invierte el orden del array.
    Ejm: 
        num = [1,2,3]
        num.reverse() = 3,2,1
